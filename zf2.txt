# this is an example zones file
# each line with parts split on white space are considered thus:
#     1:               the host
#     2:               the record type
#     everything else: either a single string or json list if it starts with "["
#     lines starting with white space are striped of white space (including "\n")
#     and added to the previous line
butts.com  A       1.2.3.4
butts.com  CNAME   whatever.com
butts.com  MX      ["whatever.com.", 5]
butts.com  MX      ["mx2.whatever.com.", 10]
butts.com  MX      ["mx3.whatever.com.", 20]
butts.com  NS      ns1.whatever.com.
butts.com  NS      ns2.whatever.com.
butts.com  TXT     hello this is some text
butts.com  SOA     ["ns1.butts.com", "dns.butts.com"]
