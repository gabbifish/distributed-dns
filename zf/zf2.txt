# this is an example zones file
# each line with parts split on white space are considered thus:
#     1:               the host
#     2:               the record type
#     everything else: either a single string or json list if it starts with "["
#     lines starting with white space are striped of white space (including "\n")
#     and added to the previous line
cats.com  A       1.2.3.4
cats.com  CNAME   petsmart.com
cats.com  MX      ["petsmart.com.", 1]
cats.com  MX      ["mx2.petsmart.com.", 2]
cats.com  MX      ["mx3.petsmart.com.", 3]
cats.com  NS      ns1.petsmart.com.
cats.com  NS      ns2.petsmart.com.
cats.com  TXT     hello this is some text
cats.com  SOA     ["ns1.cats.com", "dns.cats.com"]
